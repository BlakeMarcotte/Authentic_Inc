rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow users to read and write their own handwriting session
    match /handwriting-sessions/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow admin to read all handwriting sessions
    match /handwriting-sessions/{userId} {
      allow read: if request.auth != null && request.auth.token.email == 'admin@authenticink.com';
    }
  }
}
